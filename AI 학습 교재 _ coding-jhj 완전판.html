1.4 클래스 (Class)

클래스는 데이터(속성)와 기능(메서드)을 묶는 객체 지향 개념. PyTorch 신경망도 클래스로 정의합니다.

class Student:
    school = "AI고등학교"           # 클래스 변수 (공유)

    def __init__(self, name, age):   # 생성자
        self.name = name             # 인스턴스 변수
        self.age = age

    def introduce(self):
        return f"저는 {self.name}, {self.age}살입니다."

    def __repr__(self):              # print()할 때 출력
        return f"Student({self.name})"

    @classmethod
    def from_string(cls, s):         # "이름,나이" 문자열로 생성
        name, age = s.split(",")
        return cls(name, int(age))

    @staticmethod
    def is_adult(age):               # 인스턴스 없이 호출 가능
        return age >= 18

# 인스턴스 생성
s1 = Student("철수", 20)
s2 = Student.from_string("영희,21")
print(s1.introduce())
print(Student.is_adult(17))   # False

# 상속
class GradStudent(Student):
    def __init__(self, name, age, lab):
        super().__init__(name, age)   # 부모 생성자 호출
        self.lab = lab

    def introduce(self):             # 메서드 오버라이딩
        base = super().introduce()
        return f"{base} {self.lab} 연구실 소속."

뷰(View) vs 복사(Copy): NumPy 슬라이스는 복사본이 아닌 뷰를 반환합니다. 슬라이스를 변경하면 원본도 바뀝니다!
arr_slice = arr[5:8]

print(arr)            # [..., 12, 3450, 12, ...]

# 복사하려면 .copy() 사용
arr_slice2 = arr[5:8].copy()
arr_slice2[1] = 12    # 원본에 영향 없음

3.2 중심 경향 — 평균, 중위수, 최빈값

# ── 평균 (Mean) ──────────────────────────────
def mean(xs):
    return sum(xs) / len(xs)

print(mean(num_friends))

# ── 중위수 (Median) ───────────────────────────
def _median_even(xs):   # 짝수 개수
    sorted_xs = sorted(xs)
    high_midpoint = len(xs) // 2
    return (sorted_xs[high_midpoint - 1] + sorted_xs[high_midpoint]) / 2

def _median_odd(xs):    # 홀수 개수
    sorted_xs = sorted(xs)
    midpoint = len(xs) // 2
    return sorted_xs[midpoint]

def median(xs):
    if len(xs) % 2 == 0:
        return _median_even(xs)
    else:
        return _median_odd(xs)

print(median(num_friends))

# 이상치 영향 비교
Y = num_friends.copy()
Y[0] = 200
print(mean(Y))    # 이상치에 민감하게 바뀜
print(median(Y))  # 이상치에 강건함 (거의 안 바뀜)

# ── 사분위수 (Quantile) ───────────────────────
def quantile(xs, p):
    xs_sorted = sorted(xs)
    p_index = int(p * len(xs))
    return xs_sorted[p_index]

print("Q1:", quantile(num_friends, 0.25))
print("Q3:", quantile(num_friends, 0.75))

# ── 최빈값 (Mode) ─────────────────────────────
def mode(xs):
    counts = Counter(xs)
    max_count = max(counts.values())
    return {x_i for x_i, count in counts.items() if count == max_count}

print(mode(num_friends))

def dotV(v, w):
    assert len(v) == len(w)
    return sum(v_i * w_i for v_i, w_i in zip(v, w))

def dev_mean(xs):
    mu = mean(xs)
    return [x - mu for x in xs]

def sum_of_squares(v):
    return dotV(v, v)

def var(xs):
    assert len(xs) >= 2
    n = len(xs)
    deviations = dev_mean(xs)
    return sum_of_squares(deviations) / (n - 1)   # 불편분산 (n-1)

def corrcoef(xs, ys):
    assert len(xs) == len(ys)
    stdev_x = std(xs)
    stdev_y = std(ys)
    if stdev_x > 0 and stdev_y > 0:
        return cov(xs, ys) / (stdev_x * stdev_y)
    return 0

def vector_sum(vectors):
    assert len(vectors) > 0
    num_elements = len(vectors[0])
    assert all(len(v) == num_elements for v in vectors)
    return [sum(vector[i] for vector in vectors) for i in range(num_elements)]

    return [[entry_fn(i, j) for j in range(m)] for i in range(n)]

def addM(A, B):
    assert shape(A) == shape(B)
    m, n = shape(A)
    return make_matrix(m, n, lambda i, j: A[i][j] + B[i][j])

# 뺄셈
def subtractM(A, B):
    assert shape(A) == shape(B)
    m, n = shape(A)
    return make_matrix(m, n, lambda i, j: A[i][j] - B[i][j])

# 스칼라 곱셈
def scalar_multiplyM(c, M):
    return [[c * row_i for row_i in row] for row in M]

def matmul(A, B):
    """A: (m,n) B: (n,p) → 결과: (m,p)"""
    return [[sum(a*b for a,b in zip(A_row, B_col))
             for B_col in zip(*B)]
            for A_row in A]

A = [[2, 7], [4, 5], [7, 8]]     # 3x2
B = [[5, 8, 1, 2], [4, 5, 9, 1]] # 2x4
print(matmul(A, B))               # 3x4

M = [[3, 1], [2, 1], [1, 0]]     # 3x2
print(matmul(M, identity(2)) == M)  # True (단위행렬 곱 = 자기 자신)

# 전치행렬
def transpose(M):
    return [list(col) for col in zip(*M)]

X = [[9, 8, 7], [-1, 3, 4]]
print(transpose(X))   # [[9,-1],[8,3],[7,4]]

# 전치 법칙 검증
print(transpose(transpose(A)) == A)                              # True

7.1 퍼셉트론 & 논리 게이트

import numpy as np

def Step_Function(x):
    return 1 if x > 0 else 0

# 외출 여부 퍼셉트론
def Perceptron(x1, x2, w1, w2):
    B = -1                          # bias (기본 싫어함)
    output = x1*w1 + x2*w2 + B     # 가중합
    y = Step_Function(output)
    return output, y

# x1=비, x2=여친, w1=비 좋아하는 정도, w2=여친 좋아하는 정도
print(Perceptron(1, 0, -3, 4))   # (output, y)

# AND 게이트
def AND_gate(x1, x2):
    w = np.array([0.5, 0.5]); b = -0.7
    return Step_Function(np.sum(w * np.array([x1, x2])) + b)

# OR 게이트
def OR_gate(x1, x2):
    w = np.array([0.5, 0.5]); b = -0.2
    return Step_Function(np.sum(w * np.array([x1, x2])) + b)

# NAND 게이트 (AND의 반전)
def NAND_gate(x1, x2):
    w = np.array([-0.5, -0.5]); b = 0.7
    return Step_Function(np.sum(w * np.array([x1, x2])) + b)

for x1, x2 in [(0,0),(0,1),(1,0),(1,1)]:
    print(f"AND({x1},{x2})={AND_gate(x1,x2)}  OR({x1},{x2})={OR_gate(x1,x2)}")

# XOR = 다층 퍼셉트론으로 해결
def XOR_gate(x1, x2):
    s1 = NAND_gate(x1, x2)   # 숨은층 1
    s2 = OR_gate(x1, x2)     # 숨은층 2
    return AND_gate(s1, s2)  # 출력층

for x1, x2 in [(0,0),(0,1),(1,0),(1,1)]:
    print(f"XOR({x1},{x2}) = {XOR_gate(x1,x2)}")
# 0, 1, 1, 0 ← 완벽!

10.1 Scaled Dot-Product Attention

Attention(Q, K, V) = softmax(QKᵀ / √d_k) × V

Query, Key, Value 세 행렬로 계산. Q·Kᵀ로 유사도를 구하고, softmax로 정규화 후 V를 가중합.

import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np
import matplotlib.pyplot as plt

def scaled_dot_product_attention(Q, K, V, mask=None):
    d_k = Q.size(-1)
    scores = torch.matmul(Q, K.transpose(-2, -1)) / (d_k ** 0.5)

    if mask is not None:
        scores = scores.masked_fill(mask == 0, -1e9)

    attn_weights = F.softmax(scores, dim=-1)
    output = torch.matmul(attn_weights, V)
    return output, attn_weights

10.2 Multi-Head Self-Attention

class MultiHeadSelfAttention(nn.Module):
    def __init__(self, d_model, num_heads):
        super().__init__()
        assert d_model % num_heads == 0
        self.d_model    = d_model
        self.num_heads  = num_heads
        self.d_k        = d_model // num_heads

        self.W_q = nn.Linear(d_model, d_model)
        self.W_k = nn.Linear(d_model, d_model)
        self.W_v = nn.Linear(d_model, d_model)
        self.W_o = nn.Linear(d_model, d_model)

    def split_heads(self, x, batch_size):
        x = x.view(batch_size, -1, self.num_heads, self.d_k)
        return x.transpose(1, 2)   # (batch, heads, seq, d_k)

    def forward(self, x, mask=None):
        batch_size = x.size(0)

        Q = self.split_heads(self.W_q(x), batch_size)
        K = self.split_heads(self.W_k(x), batch_size)
        V = self.split_heads(self.W_v(x), batch_size)

        attn_output, attn_weights = scaled_dot_product_attention(Q, K, V, mask)

        # 헤드 합치기
        attn_output = attn_output.transpose(1, 2).contiguous()
        attn_output = attn_output.view(batch_size, -1, self.d_model)

        return self.W_o(attn_output), attn_weights

class ContrastiveModel(keras.Model):
    def __init__(self, image_encoder, text_encoder, **kwargs):
        super().__init__(**kwargs)
        self.image_encoder = image_encoder
        self.text_encoder  = text_encoder
        self.temperature   = 0.1

    def compile(self, optimizer, *args, **kwargs):
        super().compile(*args, **kwargs)
        self.optimizer = optimizer
        self.loss_fn   = keras.losses.SparseCategoricalCrossentropy(from_logits=True)

    def call(self, inputs):
        img_in, txt_in = inputs
        return self.image_encoder(img_in), self.text_encoder(txt_in)

    def train_step(self, data):
        (img_in, txt_in), _ = data
        with tf.GradientTape() as tape:
            img_emb, txt_emb = self((img_in, txt_in))
            logits = tf.matmul(img_emb, txt_emb, transpose_b=True)
            labels = tf.range(tf.shape(logits)[0])
            loss   = self.loss_fn(labels, logits / self.temperature)
        grads = tape.gradient(loss, self.trainable_variables)
        self.optimizer.apply_gradients(zip(grads, self.trainable_variables))
        return {"loss": loss}

11.1 Transformer 구조 개요

Self-Attention

"그 동물은 너무 피곤해서 잠들었다" → "그것"이 "동물"임을 파악. 문장 내 모든 단어가 서로를 참조.

Multi-Head Attention

여러 관점에서 동시에 Attention 계산. 문법적 관계, 의미적 관계를 동시에 학습.

Position Encoding

RNN 없이 단어 순서를 알기 위해 sin/cos 함수로 위치 정보를 벡터에 추가.

Feed Forward

Attention 후 각 위치별 독립적인 비선형 변환. 특징 추출.

11.2 Transformer Encoder (Keras)

from tensorflow import keras
import tensorflow as tf

class TransformerEncoder(keras.layers.Layer):
    def __init__(self, embed_dim, num_heads, ff_dim, rate=0.1):
        super().__init__()
        self.att = keras.layers.MultiHeadAttention(
            num_heads=num_heads, key_dim=embed_dim
        )
        self.ffn = keras.Sequential([
            keras.layers.Dense(ff_dim, activation='relu'),
            keras.layers.Dense(embed_dim)
        ])
        self.layernorm1 = keras.layers.LayerNormalization(epsilon=1e-6)
        self.layernorm2 = keras.layers.LayerNormalization(epsilon=1e-6)
        self.dropout1   = keras.layers.Dropout(rate)
        self.dropout2   = keras.layers.Dropout(rate)

    def call(self, x, training=False):
        # Self-Attention + Residual
        attn_out = self.att(x, x)
        x = self.layernorm1(x + self.dropout1(attn_out, training=training))
        # Feed Forward + Residual
        ffn_out = self.ffn(x)
        return self.layernorm2(x + self.dropout2(ffn_out, training=training))

# 텍스트 분류에 활용
embed_dim, num_heads, ff_dim, max_len, vocab_size = 64, 4, 128, 200, 20000

inputs = keras.Input(shape=(max_len,))
x = keras.layers.Embedding(vocab_size, embed_dim)(inputs)
x = TransformerEncoder(embed_dim, num_heads, ff_dim)(x)
x = keras.layers.GlobalAveragePooling1D()(x)
x = keras.layers.Dropout(0.1)(x)
x = keras.layers.Dense(32, activation='relu')(x)
outputs = keras.layers.Dense(2, activation='softmax')(x)

model = keras.Model(inputs, outputs)
model.compile(optimizer='adam', loss='sparse_categorical_crossentropy', metrics=['accuracy'])
